  10–11. 62-US / 5sim 配置示例

  # 62-US
  ENABLE_SMS=true
  SMS_PROVIDER=62us
  US62_API_KEY=你的key
  US62_GOODS_ID=12-1-7          # 留空会明确报错

  # 5sim
  SMS_PROVIDER=5sim
  FIVESIM_TOKEN=你的token
  FIVESIM_COUNTRY=england
  FIVESIM_OPERATOR=any
  FIVESIM_PRODUCT=你的product     # 留空会明确报错
  切换平台：改 SMS_PROVIDER，或单次请求传 "sms_provider":"62us"|"5sim" 覆盖。

  12–16. 调用示例（PowerShell）

  $base="http://127.0.0.1:5060"; $h=@{ "x-api-key"="change-this-register-service-key";
  "Content-Type"="application/json" }

  # 12 查询 Sub2API 分组
  Invoke-RestMethod "$base/api/sub2api/groups" -Headers $h | ConvertTo-Json -Depth 6
  # 13 查询当前短信平台状态
  Invoke-RestMethod "$base/api/sms/providers" -Headers $h | ConvertTo-Json
  Invoke-RestMethod "$base/api/sms/profile"   -Headers $h | ConvertTo-Json
  # 14 单独注册
  $b=@{url="https://...";email="test001@hotmail.com";headless=$false;timeout=180}|ConvertTo-Json
  Invoke-RestMethod "$base/api/register/start" -Method Post -Headers $h -Body $b
  # 15 单独导入
  $b=@{email="test001@hotmail.com";name="test001@hotmail.com";group_ids=@(1);concurrency=10;priority=1;enable_sm
  s=$true;sms_provider="5sim"}|ConvertTo-Json
  Invoke-RestMethod "$base/api/sub2api/import/start" -Method Post -Headers $h -Body $b
  # 16 一键注册并导入（62us；改 sms_provider 为 "5sim" 即切换）
  $b=@{url="https://...";email="test001@hotmail.com";headless=$false;timeout=300;enable_sms=$true;sms_provider="
  62us";group_ids=@(1);concurrency=10;priority=1}|ConvertTo-Json
  Invoke-RestMethod "$base/api/register-and-import/start" -Method Post -Headers $h -Body $b
